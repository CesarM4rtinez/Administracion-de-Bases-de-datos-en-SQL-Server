# Consulta SQL
$query = @"
SELECT TOP (100) [BusinessEntityID],
      [Title],

      [MiddleName],

      [Suffix],
      [JobTitle],
      [PhoneNumber],
      [PhoneNumberType],

      [AddressLine2],
      [City],
      [StateProvinceName],
      [PostalCode],
